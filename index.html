#include <iostream>
#include <vector>
#include <algorithm>

int main() {
  std::vector<int> numbers = {10, 20, 30, 40, 50};

  // Iterate and print squares
  for (int num : numbers) {
    std::cout << num * num << " ";
  }
  std::cout << std::endl;

  // Search for an element
  int search_val = 30;
  if (std::find(numbers.begin(), numbers.end(), search_val) != numbers.end()) {
      std::cout << search_val << " found." << std::endl;
  } else {
      std::cout << search_val << " not found." << std::endl;
  }

  return 0;
}

import requests

def fetch_user_data(username):
    """Fetches user data from GitHub API."""
    api_url = f"https://api.github.com/users/{username}"
    try:
        response = requests.get(api_url, timeout=5)
        response.raise_for_status() # Raises HTTPError for bad responses (4xx or 5xx)
        user_data = response.json()
        return user_data
    except requests.exceptions.RequestException as e:
        print(f"Error fetching data for {username}: {e}")
        return None

if __name__ == "__main__":
    user = "octocat"
    data = fetch_user_data(user)
    if data:
        print(f"Username: {data['login']}")
        print(f"Public Repos: {data['public_repos']}")
        print(f"Followers: {data['followers']}")
    else:
        print(f"Could not retrieve data for {user}.")
